# 範例:字串的附加 
```
#include <iostream>
#include <string>
using namespace std;

int main()
{

	string s1("Welcome");
	s1.append(" to C++"); 
	cout << s1 << endl; 
	
	string s2("Welcome");
	s2.append(" to C and Cpp", 5, 8); 
	cout << s2 << endl;
	
	string s3("Welcome");
	s3.append(" to C and Cpp", 5); 
	cout << s3 << endl; 
	
	string s4("Welcome"); 
	s4.append(14, 'G'); 
	cout << s4 << endl; 
	
	return 0;
}

```
